int main() 
{
   int number,i,sum=0;
   float per=0;

   for(i=0;i<number;i++)

       printf("Enter Your Marks for subject: %d ->  ",i+1);
       scanf("%d",&marks[i]);
   }
   
    for(int i=0;i<number;i++)

    }
   
    for(i=0;i<number;i++)

       sum+=marks[i];

}
